## 6.1 堆排序

```c++
#include <iostream>
#include <vector>

typedef int heap[];

int left(int i)
{
  return i * 2 +1;
}

int right(int i)
{
  return (i + 1) * 2;
}

int parent(int i) {
  return i / 2;
}

void swap(heap t, int i, int j)
{
  int temp = t[i];
  t[i] = t[j];
  t[j] = temp;
}

void heapfiy(heap t, int n, int i)
{
  if (i >= n)
    return;

  int l = left(i);
  int r = right(i);
  int max = i;
  if (l <= n && t[l] > t[i])
    max = l;
  if (r <= n && t[r] > t[max])
    max = r;
  if (max != i)
  {
    swap(t, max, i);
    heapfiy(t, n, max);
  }
}

int main() {
  std::cout << "Hello World!\n";
  const int n = 6;
  heap t{ 4, 10, 3, 5, 1, 2};
  heapfiy(t, n, 0);
  
  std::vector<int> arr(std::begin(t), std::end(t));
  for (int i = 0; i < n; i++)
  {
    std::cout << t[i] << std::endl;
  }

}
```
