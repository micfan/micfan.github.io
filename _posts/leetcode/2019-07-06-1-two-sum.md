class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> mp;
        for (int i = 0; i < nums.size(); i++) {
            mp[nums[i]] = i;
        }
        
        for (int i = 0; i < nums.size(); i++) {
            int expect = target - nums[i];
            
            if (mp.count(expect) && mp[expect] != i) {
                return {i, mp[expect]};
            }
        }
        
        return {0, 0};
    }
};
